#  一、队列的概念
**队列概念**：只允许在一端进行插入数据操作，在另一端进行删除数据操作的特殊线性表，队列具有**先进先出**FIFO(First In First Out) 的原则
 - **入队列**：进行插入操作的一端称为**队尾**
 - **出队列**：进行删除操作的一端称为**队头**
 下面这张图可以更好的帮助大家理解
![](https://i-blog.csdnimg.cn/direct/7b6bdd4502084fa9afcecb754d908d4b.png)
这种就像我们排队买东西，出队相当于结账完离开，入栈就是排到最后等结账
# 二、队列的结构
在我们实现它之前，我们需要先考虑它的结构来选择合适的结构来更好的实现它
我们从队列的概念中可想到用链表或数组来实现
## 选择合适的结构实现链表
这就需要我们来分析一下他们的特点，数组的话我们知道他对删除数据十分麻烦，需要把后面的元素依次向前移动，时间复杂度太高，效率太低。所以我们选择用链表实现

## 1、队列的定义
下面就是用链表来实现队列的定义,眼尖的朋友已经看见了我又用一个结构体来封装队列的头和尾已及元素个数

> 原因：
> 1、链表让实参进行改变，要传2级指针，十分麻烦，而我们封装一个结构体就会简单许多，只需修改内部成员就可以
> 2、size方便我们去用于快速获取队列长度（避免遍历链表统计）

Queue.h
```c
#pragma once
#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<assert.h>
// 类型重定义：将int定义为QDataType（Queue Data Type）
typedef int QDataType;
// 队列的节点结构体：链表的单个节点，用于存储数据并链接其他节点
typedef struct QueueNode
{
	struct QueueNode* next; // 指针域：指向后一个节点，形成链表结构
	QDataType val;          // 数据域：存储当前节点的具体数据（类型为QDataType）
}QNode; // 重命名结构体为QNode，简化后续使用
// 队列的管理结构体：封装队列的核心信息，方便操作队列
typedef struct Queue
{
	QNode* phead; // 指向队头节点（出队操作的起点）
	QNode* ptail; // 指向队尾节点（入队操作的终点）
	int size;     // 记录队列中元素的总个数，用于快速获取队列长度（避免遍历链表统计）
}Queue;
```

```c
//初始化
// 初始化队列
// 功能：将队列初始化为空状态，设置头指针、尾指针为NULL，元素个数为0
void QueueInit(Queue* pq)
{
    assert(pq); // 断言检查：确保传入的队列指针pq不为NULL（避免对空指针操作）
    pq->phead = NULL; // 队头指针初始化为NULL（空队列无节点）
    pq->ptail = NULL; // 队尾指针初始化为NULL（空队列无节点）
    pq->size = 0;     // 队列元素个数初始化为0
}
```
## 4、队列的销毁
有队列的创建就有销毁，代码如下
Queue.c
```c
// 销毁队列
// 功能：释放队列中所有节点的内存，将队列重置为空状态，避免内存泄漏
void QueueDestroy(Queue* pq)
{
    assert(pq); // 确保队列指针有效

    QNode* cur = pq->phead; // 从队头开始遍历所有节点
    while (cur) // 遍历链表，直到所有节点被释放
    {
        QNode* next = cur->next; // 先记录当前节点的下一个节点（避免释放后丢失指针）
        free(cur);               // 释放当前节点的内存
        cur = next; // 移动到下一个节点
    }
    // 重置队列状态为初始空状态（避免野指针）
    pq->phead = pq->ptail = NULL;
    pq->size = 0;
}
```

Queue.c

```c
// 队尾插入元素（入队操作）
// 功能：在队列的尾部添加一个新元素，维护队列的"先进先出"特性
void QueuePush(Queue* pq, QDataType x)
{
    assert(pq); // 确保队列指针有效
    // 1. 创建新节点并初始化
    QNode* newnode = (QNode*)malloc(sizeof(QNode)); // 为新节点分配内存
    if (newnode == NULL) // 检查内存分配是否成功（避免malloc失败导致的错误）
    {
        perror("malloc fail"); // 打印错误信息（如内存不足）
        return;
    }
    newnode->next = NULL; // 新节点的next初始化为NULL（作为队尾节点）
    newnode->val = x;     // 存储待插入的数据
    // 2. 分情况插入新节点
    if (pq->ptail == NULL) // 情况1：队列为空（头、尾指针均为NULL）
    {
        // 新节点既是队头也是队尾
        pq->phead = pq->ptail = newnode;
    }
    else // 情况2：队列非空（已有节点）
    {
        // 原队尾节点的next指向新节点，更新队尾指针为新节点
        pq->ptail->next = newnode;
        pq->ptail = newnode;
    }
    pq->size++; // 队列元素个数加1
}
```
## 6、队头删除
队列队头删除只需要将队列的地址指向第2个指针，释放第一个，元素个数size-1，代码如下
Queue.c
```c
// 队头删除元素（出队操作）
// 功能：移除队列的头部元素，维护队列的"先进先出"特性
void QueuePop(Queue* pq)
{
    assert(pq);               // 确保队列指针有效
    assert(pq->size != 0);    // 断言检查：队列不为空（避免对空队列执行出队）

    // 分情况处理：队列中只有一个节点 vs 多个节点
    if (pq->phead->next == NULL) // 情况1：队列只有一个节点（头、尾指针指向同一节点）
    {
        free(pq->phead);       // 释放唯一的节点
        pq->phead = pq->ptail = NULL; // 重置头、尾指针为NULL（队列变为空）
    }
    else // 情况2：队列有多个节点
    {
        QNode* next = pq->phead->next; // 记录头节点的下一个节点
        free(pq->phead);               // 释放当前头节点
        pq->phead = next;              // 更新头指针为下一个节点
    }
    pq->size--; // 队列元素个数减1
}

```

```c
#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<assert.h>

typedef struct QueueNode
{
	struct QueueNode* next;
	QDataType val;
}QNode;
typedef struct Queue
{
	QNode* phead;
	QNode* ptail;
	int size;
}Queue;

void QueueInit(Queue* pq);
void QueueDestroy(Queue* pq);

// 队尾插入
void QueuePush(Queue* pq, QDataType x);

QDataType QueueFront(Queue* pq);
QDataType QueueBack(Queue* pq);

bool QueueEmpty(Queue* pq);

```
Queue.c

```c
#include"Queue.h"

void QueueInit(Queue* pq)
{
	assert(pq);
	pq->phead = NULL;
	pq->ptail = NULL;
	pq->size = 0;
}

void QueueDestroy(Queue* pq)
{
	assert(pq);

	QNode* cur = pq->phead;
	while (cur)
	{
		QNode* next = cur->next;
		free(cur);

		cur = next;
	}

	pq->phead = pq->ptail = NULL;
	pq->size = 0;
}

// 队尾插入
void QueuePush(Queue* pq, QDataType x)
{
	assert(pq);

	QNode* newnode = (QNode*)malloc(sizeof(QNode));
	if (newnode == NULL)
	{
		perror("malloc fail");
		return;
	}

	newnode->next = NULL;
	newnode->val = x;

	if (pq->ptail == NULL)
	{
		pq->phead = pq->ptail = newnode;
	}
	else
	{
		pq->ptail->next = newnode;
		pq->ptail = newnode;
	}

	pq->size++;
}

// 队头删除
void QueuePop(Queue* pq)
{
	assert(pq);
	assert(pq->size != 0);

	/*QNode* next = pq->phead->next;
	free(pq->phead);
	pq->phead = next;

	if (pq->phead == NULL)
		pq->ptail = NULL;*/

		// 一个节点
	if (pq->phead->next == NULL)
	{
		free(pq->phead);
		pq->phead = pq->ptail = NULL;
	}
	else // 多个节点
	{
		QNode* next = pq->phead->next;
		free(pq->phead);
		pq->phead = next;
	}

	pq->size--;
}

QDataType QueueFront(Queue* pq)
{
	assert(pq);
	assert(pq->phead);

	return pq->phead->val;
}

QDataType QueueBack(Queue* pq)
{
	assert(pq);
	assert(pq->ptail);

	return pq->ptail->val;
}


int QueueSize(Queue* pq)
{
	assert(pq);

	return pq->size;
}

bool QueueEmpty(Queue* pq)
{
	assert(pq);

	return pq->size == 0;
}
```
