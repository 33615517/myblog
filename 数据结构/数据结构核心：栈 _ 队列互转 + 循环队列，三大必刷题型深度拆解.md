# 一、用队列实现栈
链接：[用队列实现栈](https://leetcode.cn/problems/implement-stack-using-queues/description/)
![在这里插入图片描述](https://i-blog.csdnimg.cn/direct/c1c27c3a4e404d4eb76d7c20728789f5.png)
这道题我们需要结合队列和栈的特点来写，队列是先进先出，栈是先进后出这两种结构的特性完全相反，因此 “用 A 实现 B” 的本质是用 A 的特性 “反向弥补” B 的特性，而 “辅助结构” 就是用来完成这种 “反向弥补” 的关键。我们该怎么解决呢？
## 1、思路
我们结合他们两个特点，发现单队列无法“跳过前面的元素，直接取最后一个元素”，所以我们采用两个队列来实现栈比较方便，方式为**用一个队列暂存 “非栈顶元素”，让另一个队列的最后一个元素成为可访问的栈顶**
要理解**用两个队列实现栈的核心思路**，就要分析双队列如何协作弥补这个差异。
### 双队列的协作逻辑：“转移元素暴露栈顶”
我们用**两个队列（记为`q1`和`q2`）**分工协作，核心思路是：
- 一个队列（比如`q1`）作为“**存储队列**”，保存当前所有元素；
- 另一个队列（比如`q2`）作为“**辅助队列**”，临时转移元素。

以**弹出栈顶**（`pop`操作）为例，步骤如下：
 1. 假设`q1`是存储队列（非空），`q2`是辅助队列（空）；
 2. 将`q1`中**除最后一个元素外的所有元素**，依次弹出并加入`q2`；
 3. 此时`q1`中剩下的**最后一个元素**就是**栈顶**，将其弹出并返回；
 4. 之后`q2`变为新的“存储队列”，`q1`变为辅助队列（角色互换）。
 
通过这种“转移元素”的操作，我们让队列的**队尾元素（栈顶）**被暴露出来，从而实现“后进先出”。

举个例子：
- 入栈顺序是 `1→2→3`（栈顶是`3`），此时`q1`存储`[1,2,3]`；
- 要弹出栈顶`3`，需先把`1、2`转移到`q2`，此时`q1`只剩`3`，弹出`3`即可；
- 后续`q2`存储`[1,2]`，成为新的存储队列。

```c
typedef struct Queue//队列

}Queue;
//初始化
void QueueInit(Queue* queue);
//队列是否为空
bool QueueEmpty(Queue* queue);
//进队
void QueuePush(Queue* queue, int x);
//出队
void QueuePop(Queue* queue);

//队列尾部元素

//队列有效元素个数

void QueueDestroy(Queue* queue);

{
    assert(queue);
    queue->front = NULL;
    queue->tail = NULL;
}
//队列是否为空
bool QueueEmpty(Queue* queue)
{
    assert(queue);
    return queue->tail == NULL;
}
//进队
void QueuePush(Queue* queue, int x)
{
    assert(queue);
    QNode* newnode = (QNode*)malloc(sizeof(QNode));
    assert(newnode);
    newnode->data = x;
    newnode->next = NULL;
    if(queue->tail == NULL)
    {
        queue->front = queue->tail = newnode;
    }
    else
    {
        queue->tail->next = newnode;
        queue->tail = newnode;
    }
}
//出队
void QueuePop(Queue* queue)
{
    assert(queue);
    QNode* next = queue->front->next;
    free(queue->front);
    queue->front = next;
    if(queue->front == NULL)
    {
        queue->tail = NULL;
    }
}
//队列头部元素
int QueueTop(Queue* queue)
{
    assert(queue);
    assert(queue->front);
    return queue->front->data;
}
//队列尾部元素
int QueueBack(Queue* queue)
{
    assert(queue);
    assert(queue->tail);
    return queue->tail->data;
}
//队列有效元素个数
int QueueSize(Queue* queue)
{
    assert(queue);
    int size = 0;
    QNode* cur = queue->front;
    while(cur != NULL)
    {
        ++size;
        cur = cur->next;
    }
    return size;
}
//销毁队列
void QueueDestroy(Queue* queue)
{
    assert(queue);
    QNode* next = queue->front;
    while(next != NULL)
    {
        next = queue->front->next;
        free(queue->front);
        queue->front = next;
    }
    queue->tail = NULL;
}
```
### （2）用两个队列实现栈
基于上述队列结构和我们的思路，我们定义MyStack结构体（包含两个队列），并实现栈的所有操作。
具体代码如下(详细的批注供大家理解)


```c
// 栈结构：用两个队列模拟栈（q1和q2分工协作）
// 核心逻辑：始终保持一个队列存储元素，另一个为空（辅助转移元素）
typedef struct {
    Queue q1;  // 队列1：可能存储元素或作为辅助队列
    Queue q2;  // 队列2：可能存储元素或作为辅助队列
} MyStack;

// 创建栈（初始化两个队列）
// 步骤：1. 为栈结构体分配内存 2. 初始化两个队列（确保初始状态为空）
MyStack* myStackCreate() {
    // 为栈本身分配内存
    MyStack* mystack = (MyStack*)malloc(sizeof(MyStack));
    assert(mystack);  // 确保内存分配成功（避免空指针操作）
    
    // 初始化两个队列（队列初始状态：front和tail均为NULL）
    QueueInit(&mystack->q1);
    QueueInit(&mystack->q2);
    
    return mystack;
}

// 压栈操作：将元素加入栈顶（模拟栈的"后进"特性）
// 设计思路：新元素总是加入"非空队列的队尾"，因为队尾对应栈顶
void myStackPush(MyStack* obj, int x) {
    assert(obj);  // 确保栈指针有效
    
    // 判断q1是否非空：如果q1有元素，就往q1的队尾加（新元素成为q1的队尾，即栈顶）
    if (!QueueEmpty(&obj->q1)) {
        QueuePush(&obj->q1, x);
    } else {
        // 否则往q2的队尾加（此时q2是非空队列，新元素成为q2的队尾，即栈顶）
        QueuePush(&obj->q2, x);
    }
}

// 出栈操作：移除并返回栈顶元素（模拟栈的"先出"特性）
// 核心逻辑：通过转移元素，让非空队列的最后一个元素（栈顶）暴露出来
int myStackPop(MyStack* obj) {
    assert(obj);  // 确保栈指针有效
    
    // 定义两个指针：emptyQueue（辅助队列，初始假设为q1）、nonEmptyQueue（存储元素的队列，初始假设为q2）
    Queue* emptyQueue = &obj->q1;
    Queue* nonEmptyQueue = &obj->q2;

    // 修正假设：如果q1非空，则实际nonEmptyQueue是q1，emptyQueue是q2
    if (!QueueEmpty(emptyQueue)) {
        emptyQueue = &obj->q2;
        nonEmptyQueue = &obj->q1;
    }

    // 关键步骤：将nonEmptyQueue中"除最后一个元素外的所有元素"转移到emptyQueue
    // 目的：让nonEmptyQueue中只剩下最后一个元素（即栈顶，最后入队的元素）
    while (QueueSize(nonEmptyQueue) > 1) {
        // 从nonEmptyQueue的队头弹出元素，加入emptyQueue的队尾
        QueuePush(emptyQueue, QueueTop(nonEmptyQueue));
        QueuePop(nonEmptyQueue);
    }

    // 此时nonEmptyQueue中只剩一个元素（栈顶），弹出并返回
    int top = QueueTop(nonEmptyQueue);  // 获取栈顶值
    QueuePop(nonEmptyQueue);            // 移除栈顶元素
    
    return top;
}

// 取栈顶元素：返回栈顶值（不删除）
// 设计思路：栈顶是"最后入队的元素"，而队列的队尾就是最后入队的元素，因此直接取非空队列的队尾
int myStackTop(MyStack* obj) {
    assert(obj);                  // 确保栈指针有效
    assert(!myStackEmpty(obj));   // 确保栈非空（避免访问空队列）
    
    // 如果q1非空，栈顶就是q1的队尾元素
    if (!QueueEmpty(&obj->q1)) {
        return QueueBack(&obj->q1);
    } else {
        // 否则栈顶是q2的队尾元素
        return QueueBack(&obj->q2);
    }
}

// 判断栈是否为空
// 逻辑：栈的元素只存在于其中一个队列中，因此两个队列都为空时，栈才为空
bool myStackEmpty(MyStack* obj) {
    assert(obj);  // 确保栈指针有效
    
    // 两个队列都为空 → 栈空
    return QueueEmpty(&obj->q1) && QueueEmpty(&obj->q2);
}

// 销毁栈：释放所有内存（避免内存泄漏）
// 步骤：1. 销毁两个队列（释放队列节点内存） 2. 释放栈结构体本身的内存
void myStackFree(MyStack* obj) {
    assert(obj);  // 确保栈指针有效
    
    // 销毁q1和q2（释放队列中的所有节点）
    QueueDestroy(&obj->q1);
    QueueDestroy(&obj->q2);
    
    // 释放栈结构体本身的内存
    free(obj);
    obj = NULL;  // 避免野指针（虽然函数结束后指针失效，但养成好习惯）
}
```

## 3、核心逻辑总结 
用两个队列实现栈的关键是 **“角色分工”**： 
• 始终让一个队列（nonEmptyQueue）存储元素，另一个（emptyQueue）作为辅助；
 • 压栈时，新元素加入nonEmptyQueue的队尾（队尾即栈顶）；
  • 出栈时，通过转移nonEmptyQueue的前n-1个元素到emptyQueue，暴露最后一个元素（栈顶）并弹出； 
  • 取栈顶时，直接返回nonEmptyQueue的队尾（最后入队的元素）。
    这种设计通过双队列的协作，将队列的 “先进先出” 特性转换为栈的 “后进先出” 特性，完整模拟了栈的所有操作。

我们已经写了用两个队列实现栈，现在自然能用两个栈实现队列，用两个栈后进先出的特点来实现队列先进后出的原则
## 1、思路
要清晰理解“双栈实现队列”的思路，我们可以从**角色分工、核心规则、实例演示**三个层面拆解，重点说清“新元素该加在哪”“何时转移元素”这两个关键问题，最后对比“双队列实现栈”的异同，突出两者的“形似神异”。
### （一）、双栈实现队列：核心设计与规则
#### 1. 两个栈的固定角色
- **输入栈（`pushst`）**：唯一的“接收新元素”的栈，所有`入队`操作都往这里加。  
- **输出栈（`popst`）**：唯一的“提供出队元素”的栈，所有`出队`或`取队头`操作都从这里拿。  
#### 2. 两个核心规则（保证顺序正确的关键）
- **规则1：新元素只能加入输入栈（`pushst`）**  
  原因：输出栈（`popst`）中已经是“顺序排列”的元素（栈顶是最早入队的元素），如果直接往输出栈加新元素，会打破顺序。  
  比如：输出栈已有`[1,2]`（队头是1），若直接加3到输出栈，会变成`[1,2,3]`，此时出队会先弹出3（错误，3是新元素，应该最后出队）。  

- **规则2：仅当输出栈为空时，才把输入栈的所有元素转移到输出栈**  
  原因：确保输出栈中的元素始终是“最早入队”的一批，且顺序正确。  
  转移后，输入栈清空，输出栈的元素顺序与入队顺序一致（栈顶是最早入队的元素）。  
### （二）、实例演示：一步步看元素如何流动
以“入队1→入队2→出队→入队3→出队→出队”为例，跟踪两个栈的变化：
#### 步骤1：入队1  
- 操作：往输入栈（`pushst`）加1。  
- 状态：`pushst = [1]`，`popst = []`（空）。  
#### 步骤2：入队2  
- 操作：往输入栈（`pushst`）加2（遵守规则1）。  
- 状态：`pushst = [2,1]`（栈顶是2），`popst = []`。  
#### 步骤3：出队（取队头1）  
- 检查输出栈（`popst`）：为空，触发转移（规则2）。  
- 转移：把`pushst`的元素依次弹出，压入`popst`。  
  - 弹出2→压入`popst`→`popst = [2]`；  
  - 弹出1→压入`popst`→`popst = [1,2]`（栈顶是1，即最早入队的元素）。  
- 转移后状态：`pushst = []`，`popst = [1,2]`。  
- 出队：弹出`popst`的栈顶1。  
- 最终状态：`pushst = []`，`popst = [2]`。  
#### 步骤4：入队3  
- 操作：往输入栈（`pushst`）加3（规则1：输出栈非空，仍往输入栈加）。  
- 状态：`pushst = [3]`，`popst = [2]`。  
#### 步骤5：出队（取队头2）  
- 检查输出栈（`popst`）：非空（有2），无需转移。  
- 出队：弹出`popst`的栈顶2。  
- 状态：`pushst = [3]`，`popst = []`。  
#### 步骤6：出队（取队头3）  
- 检查输出栈（`popst`）：为空，触发转移（规则2）。  
- 转移：把`pushst`的3弹出，压入`popst`→`popst = [3]`。  
- 出队：弹出`popst`的栈顶3。  
- 最终状态：`pushst = []`，`popst = []`。

### （1）构造栈的结构

```c
typedef int STDataType;
 
 typedef struct Stack
 {
 	STDataType* a;
 	int top;
 	int capacity;
 }ST;
 
 // 初始化和销毁
 void STInit(ST* pst);
 void STDestroy(ST* pst);
 
 // 入栈  出栈
 void STPush(ST* pst, STDataType x);
 void STPop(ST* pst);
 
 // 取栈顶数据
 STDataType STTop(ST* pst);
 
 // 判空
 bool STEmpty(ST* pst);
 // 获取数据个数
 int STSize(ST* pst);
 
// 初始化和销毁
 void STInit(ST* pst)
 {
 	assert(pst);
 
 	pst->a = NULL;
 	// top指向栈顶数据的下一个位置
 	pst->top = 0;
 
 	// top指向栈顶数据
 	//pst->top = -1;
 
 	pst->capacity = 0;
 }
 
 void STDestroy(ST* pst)
 {
 	assert(pst);
 
 	free(pst->a);
 	pst->a = NULL;
 	pst->top = pst->capacity = 0;
 }
 
 // 入栈  出栈
 void STPush(ST* pst, STDataType x)
 {
 	assert(pst);
 
 	// 扩容
 	if (pst->top == pst->capacity)
 	{
 		int newcapacity = pst->capacity == 0 ? 4 : pst->capacity * 2;
 		STDataType* tmp = (STDataType*)realloc(pst->a, newcapacity * sizeof(STDataType));
 		if (tmp == NULL)
 		{
 			perror("realloc fail");
 			return;
 		}
 
 		pst->a = tmp;
 		pst->capacity = newcapacity;
 	}
 
 	pst->a[pst->top] = x;
 	pst->top++;
 }
 
 void STPop(ST* pst)
 {
 	assert(pst);
 	assert(pst->top > 0);
 
 	pst->top--;
 }
 
 // 20:08继续
 // 取栈顶数据
 STDataType STTop(ST* pst)
 {
 	assert(pst);
 	assert(pst->top > 0);
 
 	return pst->a[pst->top - 1];
 }
 
 // 判空
 bool STEmpty(ST* pst)
 {
 	assert(pst);
 
 	return pst->top == 0;
 }
 
 // 获取数据个数
 int STSize(ST* pst)
 {
 	assert(pst);
 
 	return pst->top;
 }
```

### （2）实现
基于上述栈结构，我们定义MyQueue结构体（包含两个栈），并实现队列的所有操作。
代码如下：

```c
// 队列结构体：用两个栈模拟队列（核心设计：分工明确，角色固定）
// pushst：输入栈，仅负责接收新元素（入队）
// popst：输出栈，仅负责提供出队元素（出队/取队头）
typedef struct 
{
    ST pushst;  // 输入栈：所有新元素必须加入此栈，避免打乱输出栈的顺序
    ST popst;   // 输出栈：存储待出队的元素，栈顶为最早入队的元素（队头）
} MyQueue;

// 声明取队头函数（因myQueuePop依赖此函数，需提前声明）
int myQueuePeek(MyQueue* obj);

// 创建队列：初始化两个栈，确保初始状态为空
MyQueue* myQueueCreate() 
{
    // 为队列结构体分配内存
    MyQueue* obj = (MyQueue*)malloc(sizeof(MyQueue));
    // 初始化输入栈和输出栈（栈的初始化会设置top=0、capacity=0、a=NULL）
    STInit(&(obj->pushst));
    STInit(&(obj->popst));
    return obj;  // 返回初始化后的队列指针
}

// 入队操作：将元素加入队列（核心规则：新元素只能加入输入栈）
void myQueuePush(MyQueue* obj, int x) 
{
    assert(obj);  // 确保队列指针有效
    // 直接将元素压入输入栈（pushst）
    // 原因：输出栈（popst）中可能已有按顺序排列的元素，若直接加入会破坏队列顺序
    STPush(&(obj->pushst), x);
}

// 出队操作：移除并返回队头元素（依赖myQueuePeek确保队头有效）
int myQueuePop(MyQueue* obj) 
{
    // 先通过myQueuePeek获取队头元素（此时输出栈已准备好队头）
    int front = myQueuePeek(obj);
    // 弹出输出栈的栈顶（即队头元素，完成出队）
    STPop(&(obj->popst));
    return front;  // 返回出队的元素
}

// 取队头操作：返回队头元素（不删除），核心逻辑是"按需转移元素"
int myQueuePeek(MyQueue* obj) 
{
    assert(obj);  // 确保队列指针有效
    
    // 关键条件：仅当输出栈（popst）为空时，才转移输入栈（pushst）的所有元素
    // 目的：保证输出栈中的元素始终是"最早入队"的一批，且顺序正确
    if (STEmpty(&(obj->popst)))
    {
        // 将输入栈的所有元素转移到输出栈（逆序→顺序的转换）
        // 例如：输入栈为[3,2,1]（入队顺序1→2→3），转移后输出栈为[1,2,3]
        while (!STEmpty(&(obj->pushst)))  // 输入栈非空时循环转移
        {
            int top = STTop(&(obj->pushst));  // 取输入栈顶元素
            STPush(&(obj->popst), top);       // 压入输出栈（逆序变顺序）
            STPop(&(obj->pushst));            // 弹出输入栈顶，继续转移下一个
        }
    }
    
    // 此时输出栈非空，栈顶即为队头元素（最早入队的元素）
    return STTop(&(obj->popst));
}

// 判空操作：判断队列是否为空
bool myQueueEmpty(MyQueue* obj)
{
    assert(obj);  // 确保队列指针有效
    // 队列空的条件：输入栈和输出栈都为空（所有元素已出队且无新元素入队）
    // 仅一个栈空时队列非空（例如：输入栈有新元素，或输出栈有待出队元素）
    return STEmpty(&(obj->popst)) && STEmpty(&(obj->pushst));
}

// 销毁队列：释放所有内存（避免内存泄漏）
void myQueueFree(MyQueue* obj)
{
    assert(obj);  // 确保队列指针有效
    // 先销毁输入栈和输出栈（释放栈的底层数组内存）
    STDestroy(&(obj->popst));
    STDestroy(&(obj->pushst));
    // 再释放队列结构体本身的内存
    free(obj);
    // 置空指针（避免野指针，虽函数结束后失效，但为良好习惯）
    obj = NULL;
}
```

### 3、与“双队列实现栈”的对比：形似神异
| 对比维度         | 双栈实现队列                          | 双队列实现栈                          |
|------------------|---------------------------------------|---------------------------------------|
| 核心目标         | 用LIFO（栈）模拟FIFO（队列）          | 用FIFO（队列）模拟LIFO（栈）          |
| 新元素加入位置   | 固定加入“输入栈”（无论另一个栈是否为空） | 固定加入“非空队列”（始终往有元素的队列加） |
| 转移时机         | 仅当“输出栈为空”时，转移输入栈元素     | 仅当“出栈”时，转移非空队列的前n-1个元素 |
| 转移后顺序       | 输出栈元素与入队顺序一致（栈顶是队头） | 辅助队列变为新的存储队列（保留最后1个元素为栈顶） |
| 典型场景         | 入队频繁，出队均摊高效                | 入栈频繁，出栈需转移n-1个元素          |


### 总结：双栈实现队列的“不变”与“变”
- **不变**：输入栈永远负责接收新元素，输出栈永远负责提供出队元素，角色固定。  
- **变**：仅当输出栈为空时，输入栈的元素才“批量转移”到输出栈，确保顺序正确。  

这种设计通过“固定角色+延迟转移”，完美解决了“栈逆序”与“队列顺序”的矛盾，也是它与“双队列实现栈”最本质的区别——前者靠“固定分工+延迟转移”保顺序，后者靠“动态角色+即时转移”露栈顶。

## 思路
这题我们很容易联想到数组和链表来实现，但链表的特性与循环队列需求不搭，维护起来又麻烦还不划算，所以我们用数组来实现

 1.  具体思路
  我们通过两个变量，一个head指向头，一个tail指向尾的下一个，添加tail++,减少head++，因为是循环队列，所以我们当变量超过我们所设的界限，我们需要去%界限+1得到所需
但如果我们我们只开需要的空间，比如最长为n，我们开n个，这时会无法区分“空”和“满”，如图
![在这里插入图片描述](https://i-blog.csdnimg.cn/direct/7848259426324928a1acaa5d3a4f8e7d.png)
所以我们应该怎么办呢，我们可以加一个变量size，或多开一个空间，这里我们用多开一个空间
##  实现
按照题目要求，这里简单说明实现方法
 1. 构造结构体包含实现数组，head队头，tail队尾的下一个位置，k有效空间个数
 2. 初始化，已经用了很多遍了，这里不做说明
 3. isEmpty判空，tail==head
 4. isFull判满，(obj->tail + 1) % (obj->k + 1) == obj->head
 5. Front取队头数据，先判空，空返回-1，非空返回对应数组元素
 6. Rear取队尾数据，还是先判空，空返回-1，非空返回相应元素
 7. enQueue(value)插入，先判满，满返回false，不满，obj->arr[obj->tail]=value,tail=(tail+1)%(obj->k+1),返回true
 8. deQueue删除队尾元素，先判空，空返回false，非空删除，返回true
 
 **具体代码如下**
 
```c
typedef struct 
{
    int* arr; // 数组存储元素，大小为k+1（多开1个空间，用于区分“空”和“满”）
    int head; // 队头指针，指向队头元素
    int tail; // 队尾指针，指向“队尾元素的下一个位置”（此设计是区分空满的关键）
    int k;    // 循环队列的**有效容量**（用户期望的存储元素个数）
} MyCircularQueue;
 
 
MyCircularQueue* myCircularQueueCreate(int k) 
{
    MyCircularQueue* obj = (MyCircularQueue*)malloc(sizeof(MyCircularQueue));// 为队列结构体分配内存
    obj->arr = (int*)malloc(sizeof(int) * (k + 1));// 多开1个空间（k+1），解决“空”和“满”的边界混淆问题
    obj->head = 0; // 队头初始位置
    obj->tail = 0; // 队尾初始位置（指向队尾的下一个位置）
    obj->k = k; // 记录有效容量
    return obj;
}
 
bool myCircularQueueIsEmpty(MyCircularQueue* obj) 
{
    return obj->head == obj->tail; // 队头和队尾指针重合 → 队空
}
 
bool myCircularQueueIsFull(MyCircularQueue* obj) 
{
    // 队尾的“下一个位置”等于队头 → 队满（模运算处理数组环形边界）
    return (obj->tail + 1) % (obj->k + 1) == obj->head;
}
 
bool myCircularQueueEnQueue(MyCircularQueue* obj, int value)
{
    if(myCircularQueueIsFull(obj))
    {
        return false; // 队满则无法入队
    }
    else
    {
        obj->arr[obj->tail] = value; // 将元素存入队尾位置
        obj->tail++;
        obj->tail %= (obj->k + 1); // 模运算让tail绕回数组头部，实现“环形”
        return true;
    }
}
 
bool myCircularQueueDeQueue(MyCircularQueue* obj) 
{
    if(myCircularQueueIsEmpty(obj))
    {
        return false; // 队空则无法出队
    }
    else
    {
        obj->head++;
        obj->head %= (obj->k + 1); // 模运算让head绕回数组头部，实现“环形”
        return true;
    }
}
 
int myCircularQueueFront(MyCircularQueue* obj) 
{
    if(myCircularQueueIsEmpty(obj))
    {
        return -1; // 队空返回-1
    }
    else
    {
        return obj->arr[obj->head]; // 直接返回队头元素
    }
}
 
int myCircularQueueRear(MyCircularQueue* obj) 
{
    if(myCircularQueueIsEmpty(obj))
    {
        return -1; // 队空返回-1
    }
    else
    {
       // tail指向“队尾的下一个位置”，需回退一位（模运算避免负数）
       return obj->arr[(obj->tail - 1 + obj->k + 1) % (obj->k + 1)];
    }
}
 
void myCircularQueueFree(MyCircularQueue* obj) 
{
    free(obj->arr); // 释放数组内存
    free(obj); // 释放队列结构体内存
    obj = NULL; // 避免野指针
}
```
